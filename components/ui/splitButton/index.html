    ☍自动生成菜单

    必须手动引入 Controls.Menu.Menu 组件，才能自动生成菜单。

                    按钮button

                    var d1 = new SplitButton('#d1');
                    var item = d1.add('这是通过 add 添加的');
                    Dom.on(item.elem, 'click', function(){alert("菜单项点击了")});
