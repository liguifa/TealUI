    ☍JavaScript 假分页

    ❐<div id="content1"></div>
<nav class="x-pager" id="pager1"></nav>
<script>
    // 本函数仅用于文档展示。
    Doc.renderCodes();

    initPager(document.getElementById('pager1'), 136, 20, 5, 0, function(page, start, end) {
        document.getElementById('content1').innerHTML = '显示第 ' + start + ' 到 ' + end + ' 的数据';
    });
</script>
