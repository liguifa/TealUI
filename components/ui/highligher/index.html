    ☍常用 API

    手动显示工具提示
    $('[x-role="toolTip"]').role().show();
    手动关闭工具提示
    $('[x-role="toolTip"]').role().hide();
    工具提示显示后回调
    $('[x-role="toolTip"]').role().on('show', function(){ this.dom.setText('浮层被显示了') });
