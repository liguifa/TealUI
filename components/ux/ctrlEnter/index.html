    绑定Ctrl/Command+回车事件。

    ❐<input type="text" id="input" placeholder="按 Ctrl+回车">
<script>
    import ctrlEnter from "ux/ctrlEnter";
    
    ctrlEnter(input, e => {
        alert("按了 CTRL+回车");
    });
</script>
