滑动 (源码: utility/dom/swipe.js)

API 描述 示例
dom.swippable 旋转指定对象 
var elem = document.getElementById('elem');
                var log = document.getElementById('log');
                elem.swippable({
                    onSwipeStart: function () {
                        log.innerHTML = '开始滑动';
                    }, 
                    onSwipeMove: function () {
                        if (this.state === 1) {
                            log.innerHTML = '正在水平滑动：' + this.endX;
                        } else {
                            log.innerHTML = '正在垂直滑动：' + this.endY;
                        }
                    },
                    onTap: function() {
                        log.innerHTML = '按下';
                    },
                    onLongTap: function () {
                        log.innerHTML = '长按';
                    },
                    onShortTap: function () {
                        log.innerHTML = '短按';
                    },
                    onSwipeEnd: function () {
                        log.innerHTML = '滑动结束';
                    }
        });
